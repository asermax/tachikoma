```python
# In src/tachikoma/projects/hooks.py

_log = logger.bind(component="projects")

async def projects_hook(ctx: BootstrapContext) -> None:
    """Bootstrap hook: create projects dir and sync all submodules."""
    workspace_path = ctx.settings_manager.settings.workspace.path
    projects_dir = workspace_path / "projects"

    # Create projects dir (idempotent)
    projects_dir.mkdir(exist_ok=True)

    # Discover submodules
    submodule_paths = await list_submodules(workspace_path)
    if not submodule_paths:
        return  # No-op

    # Sync in parallel with error isolation
    results = await asyncio.gather(
        *[_sync_submodule(workspace_path, path) for path in submodule_paths],
        return_exceptions=True,
    )

    # Log failures
    for path, result in zip(submodule_paths, results):
        if isinstance(result, Exception):
            _log.warning("Submodule sync failed: path={path} err={err}", path=path, err=str(result))

# _sync_submodule: wraps _sync_once with 1 retry
# _sync_once: init → fetch → resolve default branch → checkout → pull
```
